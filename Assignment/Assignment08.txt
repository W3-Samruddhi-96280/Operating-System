#!/bin/bash

#----------------------------------------
#  Greatest of 3 nos    
#---------------------------------------

read num1 num2 num3

max=0

if [ $num1 -eq $num2 ]
then
        echo "Num and num2 are equal "
        max=$num1
elif [ $num1 -gt $num2 ] && [ $num1 -gt $num3 ]
then
        echo "num1 is greater "
        max=$num1
elif [ $num2 -gt $num1 ] && [ $num2 -gt $num3 ]
then
        echo "Num2 is greater"
        max=$num2
else
        echo "num3 is greater"
        max=$num3
fi
echo "Maximum value =$max"

#!/bin/bash

#----------------------------------------
#   Factorial of Given no
#---------------------------------------



factorial ()
{
 fact=1
 for (( i=1 ; i <= $1 ; i++ ))
 do
         fact=$((fact * i))
done
echo $fact

}

fact=$(factorial 8)
echo "8! = $fact"
~                                                                                                        
~
